单链表基本操作

#include<stdio.h>
#include<stdlib.h>
#include<iostream>
using namespace std;

//定义单链表结点结构体

typedef int ElemType;

typedef struct Node
{

ElemType data;

struct Node *next;

}LNode,*LinkList;

//建立一个带头结点的单链表

void Build(LinkList L)
{

int n; 
LinkList p,q;

p=L;

printf("输入n和n个数字：\n");

scanf("%d",&n);

while(n--)
{

    q=(LinkList)malloc(sizeof(LNode));

    scanf("%d",&q->data);

    q->next=NULL;

    p->next=q;

    p=q;
}}

//求单链表的长度

void LinkLength(LinkList L)

{

int num = 0;

LinkList p;

p=L->next;

while(p)
{

    num++;

    printf("%3d",p->data);

    p=p->next;

}
printf("长度为：%d\n",num);}

//查找值为x的元素值q

void Find(LinkList L,int x)
{

LinkList p;

p=L;

int i,b=1,a;

while(p->next &&p->next->data!=x)

    {
    p=p->next;

    b++;
    }

if(p->next)
    {

    printf("%d的p位置为：%d\n\n",x,b);

    printf("是否需要修改值 1或0\n");

    cin>>a;

    if(a==1)
    {

    printf("修改的值为\n"); 

    cin>>i;

    p->next->data=i;

    printf("修改成功\n\n");

    }
else

    printf("取消修改!!\n\n");

    }
else

printf("没找到!!\n\n");} 

//插入值为x的结点

void ListInsert(LinkList L,int x)
{
LinkList p,q;

p=L;

int b;//插入的位置 

while(p->next&&p->next->data==x)

    p=p->next;

if(p->next)
{

    printf("请输入要插入的位置\n");

    cin>>b;

    for(int i=0;i<b-1;i++)
    {

        p=p->next;

    }
    q=new LNode;

    q->data=x;

    q->next=p->next;

    p->next=q;

    printf("插入成功!!\n\n",x);}}

//删除值为x的结点

void Delete(LinkList L,int x)
{

LinkList p,q;

p=L;

int b=0;//删除的位置 

while(p->next &&b<x-1)
{

        p=p->next;

        ++b;
}

if(p->next)
{ 

    q=p->next;

    p->next=q->next;

    free(q);

    printf("删除成功!!\n\n",x);}}

void Tips()
{

printf("\n");

printf("根据按键选择相应的操作：\n");

printf("<1>输出单链表的数据元素:\n");

printf("<2>查找修改元素:\n");

printf("<3>插入结点:\n");

printf("<4>删除结点:\n");

printf("<0>退出:\n");

}
主函数

int main()
{

int op,x;
LinkList L;
L = (LinkList)malloc(sizeof(LNode));
L->next=NULL;
L->data=-1;
Build(L);
Tips();
scanf("%d",&op);
while(op)
{
    switch(op)
    {
    case 1:LinkLength(L);
        break;
    case 2:
        printf("查找和修改:\n");
        scanf("%d",&x);
        Find(L,x);
        break;
    case 3:
        printf("插入的值:\n");
        scanf("%d",&x);
        ListInsert(L,x);
        break;
    case 4:
        printf("删除位置:\n");
        scanf("%d",&x);
        Delete(L,x);
        break;
    }
        scanf("%d",&op);
}
return 0;

}
