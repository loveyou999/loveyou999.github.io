素性检验之Eratosthenes筛法(埃式筛法) 

#include <stdio.h>
#include <iostream>
using namespace std;
int main() {
    int n;
    cin>>n;
    int i = 0; // 表示整数和对应的下标
    int j = 0; // 表示正在处理的j的倍数从2开始到j * k < SIZE.
    int flag[n] = {0}; // 下标表示整数内容,标识是否是质数.
    flag[0] = flag[1] = 0; // 0和1不是质数.
    for(i=2;i<=n;++i )
    { // 除了0和1其它默认全是质数.
    flag[i] = 1;
    }
    // 处理数的倍数.
    for(i=0;i<=n;++i) 
    {
        if(1==flag[i]) 
        {
            for(j=2;i*j<=n;++j) 
            {
                flag[i*j]=0;
            }        
        }
    }
    // 打印质数.
    /*for(i=0;i<=n;++i) 
    {
        if(1==flag[i]) 
        {
            cout<<i<<endl;
        }
    }*/
    if(flag[n]==1)
        cout<<"是素数"<<endl;
    else
        cout<<"不是素数"<<endl;
    return 0;
}
