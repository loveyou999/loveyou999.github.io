素性检验之MillerRabbin(米勒-拉宾) 

#include <iostream>
#include <cstring>
#include <cstdlib>
using namespace std;
long long mulmod(long long a, long long b, long long mod)//求a*b%mod 
{
    long long x=0,y=a%mod;
    while(b>0)
    {
        if(b%2==1)
        {    
            x=(x+y)%mod;
        }
        y=(y+y)%mod;
        b=b/2;
    }
    return x%mod;
}
//base^e (mod p)
long long mod1(long long base, long long e, long long mod)//base是随机产生，e是n-1/2，mod是p 
{
    long long x=1;
    long long y=base;
    while(e>0)
    {
        if(e%2==1)//e为奇数 
            x=(x*y)%mod;
        y=(y*y)%mod;
        e=e/2;
    }
    return x%mod;
}
bool Miller(long long n)
{
    if (n==2)
        return true;
    if (n<2||!(n&1))
        return false;
    long long m=n-1,k=0;
    while(!(m&1))
    {
        k++;
        m>>=1;
    }
    for(int i=1;i<=20;i++)  // 20为Miller-Rabin测试的迭代次数
    {
        long long a=rand()%(n-1)+1;//随机生成1-n之间的数 
        long long x=mod1(a,m,n);
        long long y;
        for (int j=1;j<=k;j++)
        {
            y=mulmod(x,x,n);
            if(y==1&&x!=1&&x!=n-1)
                return false;
            x=y;
        }
        if (y!=1)
            return false;
    }
    return true;
}
//Main
int main()
{
    long long n;
    cout<<"输入要判断的数:"<<endl;
    cin>>n;
    if (Miller(n))
        cout<<n<<"是素数"<<endl;
    else
        cout<<n<<"不是素数"<<endl;
    return 0;
}
