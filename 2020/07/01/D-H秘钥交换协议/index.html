D-H秘钥交换协议

1.用rand库：

#include<iostream>
#include<time.h>  
#include<stdlib.h>  
#include<math.h>
#include<cstdio>
using namespace std; 
long long mulmod(long long a, long long b, long long mod)
{
    long long x=0,y=a%mod;
    while(b>0)
    {
        if(b%2==1)
        {    
            x=(x+y)%mod;
        }
        y=(y+y)%mod;
        b=b/2;
    }
    return x%mod;
}
//base^e (mod p)
long long mod1(long long base, long long exponent, long long mod)//base是随机产生，exponent是 p-1或者p-1/2，mod是p 
{
    long long x=1;
    long long y=base;
    while(exponent>0)
    {
        if(exponent%2==1)
            x=(x*y)%mod;
        y=(y*y)%mod;
        exponent=exponent/2;
    }
    return x%mod;
}
bool Miller(long long p)
{//p是要判断的数 
    if(p<2)
    {
        return false;
    }
    if(p!=2&&p%2==0)
    {
        return false;//2的倍数 
    }
    long long s=p-1;
    while(s%2==0)
    {
        s=s/2;
    }
    for(int i=0;i<5;i++)//因为算法不是确定性，是概率性的，多运行几次，判断成功率高 
    {
        long long a=rand()%(p-1)+1,temp=s;//随机产生一个1到p-1之间的数
        long long mod=mod1(a,temp,p);//a^s (mod p)
        while(temp!=p-1&&mod!=1&&mod!=p-1)//二次探测 
        {
            mod=mulmod(mod,mod,p);
            temp=2*temp;
        }
        if(mod!=p-1&&temp%2==0)
        {
            return false;//返回合数 
        }
    }
    return true;
}
long long powm(long long base, long long a,long long md) 
{
    long long ans=1;
    while (a!=0) 
    {
        if (a&1)    
            ans=ans*base%md;
        base=base*base%md;
        a >>= 1;
    }
    return ans;
}

int main()  
{  
    unsigned long p,a,XA,XB,YA,YB,KA,KB;
    bool flag=true;
    while(flag){
    p=rand()%10000000000000+10000000000;
    if(Miller(p))
        {
        cout<<"素数P:"<<p<<endl;
        flag=false;
        }
    //else 
        //cout<<p<<"不是素数"<<endl;
    }
    a=p-1;//欧拉函数
    flag=true;
    while(flag){
    XA=rand()%10000000000000+10000000000;
    if(Miller(p)&&XA<p)
        {
        cout<<"A私钥:"<<XA<<endl;
        flag=false;
        }
    //else 
        //cout<<XA<<"不是素数"<<endl;
    }
    flag=true;
    while(flag){
    XB=rand()%10000000000000+10000000000;
    if(Miller(p)&&XB<p&&XA!=XB)
        {
        cout<<"B私钥:"<<XB<<endl;
        flag=false;
        }
    //else 
        //cout<<XB<<"不是素数"<<endl;
    }
    long long g=rand()%(p-1)+1;
         while(powm(g,a,p)!=1)
         {
             g=rand()%(p-1)+1;
        }
    cout<<"其中一个原根"<<g<<endl;
    YA=mod1(g,XA,p);
    YB=mod1(g,XB,p);
    KA=mod1(YB,XA,p);
    KB=mod1(YA,XB,p);
    cout<<"A公开密钥:"<<YA<<endl;
    cout<<"B公开密钥:"<<YB<<endl;
    cout<<"A生成的共享秘钥:"<<KA<<endl;
    cout<<"B生成的共享秘钥:"<<KB<<endl;
    return 0;  
}

2.用gmp库（大数生成）:

#include <iostream>
#include <gmp.h>
using namespace std;

int main()
{        
  mpz_t a,p,XA,XB,YA,YB,KA,KB,n; 
  mpz_init(a);
  mpz_init(p);
  mpz_init(XA);
  mpz_init(XB);
  mpz_init(YA);
  mpz_init(YB);
  mpz_init(KA);
  mpz_init(KB);

  clock_t time = clock();

gmp_randstate_t grt;
gmp_randinit_default(grt);
gmp_randseed_ui(grt, time);

mpz_urandomb(p,grt, 1024);
mpz_nextprime(p,p);

do{
mpz_urandomb(XA, grt,1024);
mpz_nextprime(XA,XA);
}while(mpz_cmp(XA,p)>=0);

do{
mpz_urandomb(XB, grt, 1024);
  mpz_nextprime(XB,XB);
}while(mpz_cmp(XB,p)>=0);

gmp_printf("%Zd\n", p);
gmp_printf("%Zd\n", XA);
gmp_printf("%Zd\n", XB);

mpz_init_set(a,p-1);
mpz_powm(YA,a,XA,p);
gmp_printf("%s %ZX\n\n", "YA",YA);
mpz_powm(YB,a,XB,p);
gmp_printf("%s %ZX\n\n", "YB",YB);
mpz_powm(KA,YB,XA,p);
gmp_printf("%s %ZX\n\n", "KA",KA);
mpz_powm(KB,YA,XB,p);
gmp_printf("%s %ZX\n\n", "KB",KB);


  mpz_clear(a);//　清空内存
  mpz_clear(p);
  mpz_clear(XA);
  mpz_clear(XB);
  mpz_clear(YA);
  mpz_clear(YB);
  mpz_clear(KA);
  mpz_clear(KB);
  mpz_clear(n);
  return 0;  
}
